  solvers:
    - dns01:
        webhook:
          groupName: {{ .Values.groupName }}
          solverName: infoblox-wapi
          config:
            host: <infoblox-host>
            view: <dns-view>
            usernameSecretRef:
              name: infoblox-credentials
              key: username
            passwordSecretRef:
              name: infoblox-credentials
              key: password
